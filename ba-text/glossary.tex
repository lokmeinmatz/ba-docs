\newglossaryentry{greenfield}{
  name=Greenfield development,
  description={, the opposite of $\rightarrow$Brownfield development, refers to software developed in a new environment without legacy code \cite{JohnAdamsIt:Greenfield}}
}

\newglossaryentry{brownfield}{
  name=Brownfield development,
  description={, the opposite of $\rightarrow$Greenfield development, refers to adding ''new capabilities on an existing product or product platform, using existing technology'' \cite{BrownfieldToGreenfield:2021ys}}
}

\newglossaryentry{saas}{
  name=SaaS,
  description={(Software as a Service) is a software distribution model where a provider hosts the software on own servers (or in the cloud) and provides the users access via the internet}
}

\newglossaryentry{devops}{
  name=DevOps,
  description={is a set of practices, tools and development culture to reduce time and friction between writing software and deploying it to production, often utilizing automated workflows}
}

\newglossaryentry{localstorage}{
  name=LocalStorage,
  description={is an API to a persistant key-value storage in the browser. It is scoped per domain, so no external pages can access the contents, but is persistent between browser sessions}
}

\newglossaryentry{json-schema}{
  name=JSON Schema,
  description={is a specification and a declarative language "that allows you to annotate and validate JSON documents." (Retrieved 2023-01-11, from \url{https://json-schema.org/})}
}

\newglossaryentry{comp-over-inh}{
  name=Composition over Inheritance,
  description={is a principle originating from object oriented programming to instances of classes instead of extending them so they are more loosely coupled. React, while beeing functional, has a similar pattern: \url{https://reactjs.org/docs/composition-vs-inheritance.html}}
}

\newglossaryentry{design-thinking}{
  name=Design Thinking,
  description={is an approach to solve problems or develop new ideas through experimentation and iteration, focused on creativity and collaboration}
}

\newglossaryentry{ux}{
  name=UX,
  description={describes all the impressions a user has when interacting with the product}
}

\newglossaryentry{experience}{
  name=Purple Experience,
  description={is a proprietary meta framework built on top of Angular for the use in apps and websites for publishers of magazines and news}
}

\newglossaryentry{manager}{
  name=Purple Manager,
  description={provides functionality to create and configure Purple apps, their content, entitlement systems and more settings related to app and website management}
}

\newglossaryentry{rest}{
  name=REST API,
  description={is an Web (HTTP) API paradigma to provide consumers of the API an standardized and esay to access endpoint structure and structured response data in a common machine-readable pattern}
}

\newglossaryentry{kubernetes}{
  name=Kubernetes,
  description={is an open-source container orchestration system for automating the deployment, scaling and management of containerized applications}
}

\newglossaryentry{ide}{
  name=IDE,
  description={, short for Integrated Development Environment, is a common application used in software development, build around a text editor combined with tools for analyzing and distributing the software}
}

\newglossaryentry{lean-ux}{
  name=Lean UX,
  description={apporaches development of good user expeirences through fast iteration and quick validation of design decisions}
}

\newglossaryentry{agile}{
  name=Agile,
  description={is a methodology for software development based on iterative and incremental delivery, providing flexibility and rapid response to change}
}
